SMART CONTRACT ARCHITECTURE
================================================================================

CONTRACT NAME: WorldBankReserve.sol

STATE VARIABLES
--------------------------------------------------------------------------------
• address public admin
• uint256 public totalReserve
• uint256 public loanCounter
• mapping(uint256 => Loan) public loans
• mapping(address => uint256[]) public userLoans
• mapping(address => uint256) public userDeposits

STRUCTURES
--------------------------------------------------------------------------------
struct Loan {
    uint256 id;
    address borrower;
    uint256 amount;
    string purpose;
    LoanStatus status;
    uint256 requestedAt;
    uint256 approvedAt;
}

enum LoanStatus {
    Pending,
    Approved,
    Rejected,
    Paid
}

CORE FUNCTIONS
--------------------------------------------------------------------------------

1. depositToReserve() external payable
   - Accepts ETH deposits
   - Updates totalReserve
   - Records user contribution
   - Emits ReserveDeposited event

2. requestLoan(uint256 amount, string memory purpose) external
   - Creates loan request
   - Validates amount <= totalReserve
   - Assigns unique loan ID
   - Emits LoanRequested event

3. approveLoan(uint256 loanId) external onlyAdmin
   - Validates loan exists and is pending
   - Transfers funds to borrower
   - Updates loan status
   - Emits LoanApproved event

4. rejectLoan(uint256 loanId) external onlyAdmin
   - Updates loan status to rejected
   - Emits LoanRejected event

5. getLoansByUser(address user) external view returns (uint256[])
   - Returns array of loan IDs for user

6. getTotalReserve() external view returns (uint256)
   - Returns current reserve balance

7. getPendingLoans() external view returns (Loan[])
   - Returns array of pending loans (admin only)

EVENTS
--------------------------------------------------------------------------------
event ReserveDeposited(address indexed depositor, uint256 amount, uint256 timestamp)
event LoanRequested(uint256 indexed loanId, address indexed borrower, uint256 amount)
event LoanApproved(uint256 indexed loanId, address indexed borrower, uint256 amount)
event LoanRejected(uint256 indexed loanId, address indexed borrower)

MODIFIERS
--------------------------------------------------------------------------------
modifier onlyAdmin() {
    require(msg.sender == admin, "Only admin can call this function");
    _;
}

SECURITY FEATURES
--------------------------------------------------------------------------------
• ReentrancyGuard for fund transfers
• Access control for admin functions
• Input validation on all functions
• SafeMath for arithmetic operations
• Emergency pause functionality

QR CODE INTEGRATION SPECIFICATION
================================================================================

QR CODE TYPES
--------------------------------------------------------------------------------

1. WALLET ADDRESS QR
   Format: ethereum:<address>
   Usage: Share wallet for receiving deposits
   Example: ethereum:0x742d35Cc6634C0532925a3b844Bc9e7595f0bEb9

2. TRANSACTION REQUEST QR
   Format: ethereum:<contract>@<chainId>/transfer?address=<recipient>&uint256=<amount>
   Usage: Pre-filled transaction requests
   Example: ethereum:0x123...@80001/depositToReserve?uint256=1000000000000000000

3. LOAN DETAILS QR
   Format: https://<domain>/loan/<loanId>
   Usage: Share loan information
   Example: https://cryptoworldbank.com/loan/42

4. PAYMENT QR (EIP-681 Standard)
   Format: ethereum:<address>@<chainId>?value=<amount>&gas=<gasLimit>
   Usage: Direct payment requests
   Example: ethereum:0x123...@80001?value=1e18&gas=21000
